`; renderCustomers(); } function addCustomer(){ let name = document.getElementById("c_name").value.trim(); let phone = document.getElementById("c_phone").value.trim(); if(!name) return alert("نام لازم است"); DB.customers.push({ id: uid(), name, phone, balance: 0 }); saveDB(); renderCustomers(); document.getElementById("c_name").value=""; document.getElementById("c_phone").value=""; } function renderCustomers(){ let box = document.getElementById("customersList"); if(!box) return; box.innerHTML = DB.customers.map(c=>`

`).join(""); } // ========================= // PAYMENT / SETTLEMENT // ========================= function openPay(id){ let c = DB.customers.find(x=>x.id===id); let html = `

تسویه حساب

${c.name}
`; document.getElementById("reports").innerHTML = html; nav("reports", document.querySelector(".bottom button:nth-child(5)")); } function pay(id){ let amount = Number(document.getElementById("pay_amount").value); if(!amount || amount<=0) return alert("مقدار نامعتبر"); let c = DB.customers.find(x=>x.id===id); c.balance -= amount; DB.ledger.push({ id: uid(), type:"payment", customerId:id, amount, date:new Date().toISOString() }); saveDB(); renderCustomers(); renderReports(); alert("تسویه ثبت شد ✔"); } // ========================= // LEDGER HELP // ========================= function addDebt(customerId,amount){ let c = DB.customers.find(x=>x.id===customerId); if(!c) return; c.balance += amount; DB.ledger.push({ id:uid(), type:"debt", customerId, amount, date:new Date().toISOString() }); saveDB(); } // ========================= // ITEMS + INVOICE ENGINE // ========================= // ------------------------- // ITEMS // ------------------------- function addItem(){ let name = document.getElementById("i_name").value.trim(); let stock = Number(document.getElementById("i_stock").value || 0); let wage = Number(document.getElementById("i_wage").value || 0); if(!name) return alert("نام کالا لازم است"); DB.items.push({ id: uid(), name, stock, wage }); saveDB(); renderItems(); document.getElementById("i_name").value=""; document.getElementById("i_stock").value=""; document.getElementById("i_wage").value=""; } function renderItems(){ let box = document.getElementById("itemsList"); if(!box) return; box.innerHTML = DB.items.map(i=>`

`).join(""); } // ------------------------- // INVOICE ENGINE // ------------------------- let invoiceRows = []; function addRow(){ invoiceRows.push({ itemId:null, weight:0, wage:0 }); renderInvoice(); } function renderInvoice(){ let box = document.getElementById("invoice"); if(!box) return; box.innerHTML = `

${invoiceRows.map((r,i)=>`
`).join("")}
`; } function setItem(i,id){ let item = DB.items.find(x=>x.id==id); invoiceRows[i].itemId = id; invoiceRows[i].wage = item ? item.wage : 0; renderInvoice(); } // ------------------------- // SAVE INVOICE (CORE LOGIC) // ------------------------- function saveInvoice(){ let cid = document.getElementById("inv_customer").value; if(!cid) return alert("مشتری انتخاب نشده"); if(invoiceRows.length===0) return alert("هیچ ردیفی نیست"); // check stock for(let r of invoiceRows){ let item = DB.items.find(i=>i.id==r.itemId); if(!item) return alert("کالا نامعتبر"); if(item.stock < r.weight){ return alert("موجودی کم است: " + item.name); } } // invoice number let number = DB.settings.invoiceNumber++; // process invoice let total = 0; invoiceRows.forEach(r=>{ let item = DB.items.find(i=>i.id==r.itemId); item.stock -= r.weight; // debt logic addDebt(Number(cid), r.weight); total += r.weight; }); // save invoice DB.invoices.push({ id: uid(), number, customerId: Number(cid), rows: invoiceRows, total, date: new Date().toISOString() }); invoiceRows = []; saveDB(); renderItems(); renderCustomers(); renderInvoice(); alert("فاکتور ثبت شد ✔"); } // ========================= // REPORTS + INVOICE LIST + EXPORT (FINAL CORE) // ========================= // ------------------------- // REPORTS // ------------------------- function renderReports(){ let box = document.getElementById("reports"); if(!box) return; let totalCustomers = DB.customers.length; let totalItems = DB.items.length; let totalStock = DB.items.reduce((s,i)=>s+i.stock,0); let totalInvoices = DB.invoices.length; let totalDebt = DB.customers.reduce((s,c)=>s+c.balance,0); box.innerHTML = `

گزارش کلی

بدهی کل: ${fix(totalDebt)} گرم

فاکتورهای اخیر

${DB.invoices.slice().reverse().map(inv=>{ let c = DB.customers.find(x=>x.id==inv.customerId); return `
#${inv.number}
${c ? c.name : "-"}
${fix(inv.total)} گرم
`; }).join("")}
`; } // ------------------------- // BACKUP // ------------------------- function backup(){ let data = JSON.stringify(DB); let blob = new Blob([data], {type:"application/json"}); let url = URL.createObjectURL(blob); let a = document.createElement("a"); a.href = url; a.download = "gold-v33-backup.json"; a.click(); } // ------------------------- // RESTORE // ------------------------- function restore(e){ let file = e.target.files[0]; let reader = new FileReader(); reader.onload = function(){ DB = JSON.parse(reader.result); saveDB(); renderCustomers(); renderItems(); renderInvoice(); renderReports(); alert("ریستور شد ✔"); }; reader.readAsText(file); } // ------------------------- // INIT FIX // ------------------------- function startApp(){ loadDB(); renderCustomers(); renderItems(); renderInvoice(); renderReports(); } startApp(); // ========================= // FINAL POLISH + APP FIX + WHATSAPP + PDF EXPORT (V33 FINAL) // ========================= // ------------------------- // WHATSAPP SHARE INVOICE // ------------------------- function shareInvoice(invId){ let inv = DB.invoices.find(i=>i.id==invId); if(!inv) return; let c = DB.customers.find(x=>x.id==inv.customerId); let text = `فاکتور شماره ${inv.number} مشتری: ${c ? c.name : "-"} مجموع: ${fix(inv.total)} گرم تاریخ: ${new Date(inv.date).toLocaleDateString("fa-IR")}`; let url = "https://wa.me/?text=" + encodeURIComponent(text); window.open(url, "_blank"); } // ------------------------- // SIMPLE PDF EXPORT (PRINT MODE) // ------------------------- function printInvoice(invId){ let inv = DB.invoices.find(i=>i.id==invId); if(!inv) return; let c = DB.customers.find(x=>x.id==inv.customerId); let w = window.open(""); w.document.write(` Invoice

`); w.print(); } // ------------------------- // DELETE INVOICE (SAFE) // ------------------------- function deleteInvoice(id){ if(!confirm("فاکتور حذف شود؟ موجودی و بدهی برمی‌گردد")) return; let invIndex = DB.invoices.findIndex(i=>i.id==id); if(invIndex==-1) return; let inv = DB.invoices[invIndex]; // rollback stock + debt inv.rows.forEach(r=>{ let item = DB.items.find(i=>i.id==r.itemId); if(item) item.stock += r.weight; let c = DB.customers.find(x=>x.id==inv.customerId); if(c) c.balance -= r.weight; }); DB.invoices.splice(invIndex,1); saveDB(); renderCustomers(); renderItems(); renderReports(); alert("فاکتور حذف شد ✔"); } // ------------------------- // SAFE GUARDS // ------------------------- window.addEventListener("beforeunload", ()=>{ saveDB(); }); // ------------------------- // FINAL INIT PATCH // ------------------------- function boot(){ loadDB(); renderCustomers(); renderItems(); renderInvoice(); renderReports(); } boot();
